\begin{document}
% start the recipe with the \recipe command
\recipe{Dill Dip}
% set the formatting for the first page only
\thispagestyle{firstpage}
% sign every page with name and url
\sign{Patrick Cook}{https://github.com/pdcook/Recipes}

% begin writing the recipe
\ingredients
\begin{multicols}{2}
\begin{itemize}
    \item 2/3 cup Mayonnaise
    \item 2/3 cup Sour Cream
    \item 2 tsp Dried Dill Weed
    \item 1 tsp Dried Parsley Flakes
    \item 1 tsp Dried Onion Flakes
    \item 1 tsp Beau Monde
    \item 1 loaf Jewish Rye Bread or Seedless Rye
\end{itemize}
\columnbreak
\end{multicols}
\instructions
Procure a medium-large bowl with a lid, make space in refrigerator.
\begin{enumerate}
    \item Using a spoon, mix all ingredients (except bread) in the bowl.
    \item Chill for several hours.
    \item Serve with bread to dip.
\end{enumerate}
